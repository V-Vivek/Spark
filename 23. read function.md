# PySpark Read Functions


## 1. READ CSV FILES
```python
df_csv = spark.read.csv(
    "path/to/file.csv",
    header=True,              # First row as column names
    inferSchema=True,         # Automatically infer/identify data types
    sep=",",                  # Delimiter (default is comma)
    quote='"',                # Quote character
    escape="\\",              # Escape character
    nullValue="NULL",         # String representation of null
)
```

## 2. READ TEXT FILES

```python
df_text = spark.read.text(
    "path/to/directory_or_file",
    wholetext=False,            # Read each file as single row if True
    recursiveFileLookup=False   # Recursive directory lookup
)
```

# 3. READ JSON FILES

```python
df_json = spark.read.json(
    "path/to/file.json",
    allowComments=False,      # Allow Java/C++ style comments
    allowUnquotedFieldNames=False, 
    allowSingleQuotes=True,   # Allow single quotes
    allowNumericLeadingZeros=False
)
```

# 4. READ PARQUET FILES

```python
df_parquet = spark.read.parquet(
    "path/to/file.parquet",
    mergeSchema=False,        # Merge schemas from multiple files
    recursiveFileLookup=False,
    modifiedBefore=None,      # Filter by modification time
    modifiedAfter=None
)
```

# 5. READ ORC FILES

```python
df_orc = spark.read.orc(
    "path/to/file.orc",
    mergeSchema=False,        # Merge schemas from multiple files
    pathGlobFilter=None,
    recursiveFileLookup=False,
    modifiedBefore=None,
    modifiedAfter=None
)
```

# 6. READ AVRO FILES

```python
df_avro = spark.read.format("avro").load(
    "path/to/file.avro",
    # Options
    avroSchema=None,          # JSON string of Avro schema
    recordName="topLevelRecord",
    recordNamespace="",
    ignoreExtension=True,     # Ignore .avro extension
    compression="snappy"      # none, deflate, snappy, bzip2, xz
)

# Alternative with options
df_avro_alt = spark.read.format("avro") \
    .option("ignoreExtension", "true") \
    .load("path/to/file.avro")
```

# 7. READ FROM HIVE TABLES

```python
# Enable Hive support in SparkSession
spark_with_hive = SparkSession.builder \
    .appName("Hive Read") \
    .enableHiveSupport() \
    .getOrCreate()

# Read Hive table
df_hive = spark_with_hive.table("database_name.table_name")

# Or using SQL
df_hive_sql = spark_with_hive.sql("SELECT * FROM database_name.table_name")

# Read with partition filter
df_hive_partition = spark_with_hive.sql(
    "SELECT * FROM database_name.table_name WHERE year=2023 AND month=12"
)
```

# 8. READ FROM JDBC (Databases)

```python
df_jdbc = spark.read.jdbc(
    url="jdbc:postgresql://hostname:5432/database",
    table="schema.table_name",
    properties={
        "user": "username",
        "password": "password",
        "driver": "org.postgresql.Driver"
    }
)

# JDBC with additional options
df_jdbc_advanced = spark.read.format("jdbc") \
    .option("url", "jdbc:mysql://hostname:3306/database") \
    .option("dbtable", "table_name") \
    .option("user", "username") \
    .option("password", "password") \
    .option("driver", "com.mysql.jdbc.Driver") \
    .option("fetchsize", "1000") \
    .option("numPartitions", "4") \
    .option("partitionColumn", "id") \
    .option("lowerBound", "1") \
    .option("upperBound", "100000") \
    .option("sessionInitStatement", "SET TIME ZONE 'UTC'") \
    .option("pushDownPredicate", "true") \
    .load()

# JDBC with custom query
df_jdbc_query = spark.read.jdbc(
    url="jdbc:postgresql://hostname:5432/database",
    table="(SELECT * FROM table WHERE condition) as subquery",
    properties={
        "user": "username",
        "password": "password",
        "driver": "org.postgresql.Driver"
    }
)
```

# 9. GENERIC FORMAT READ (Works with all formats)

```python
df_generic = spark.read \
    .format("csv") \
    .option("header", "true") \
    .option("inferSchema", "true") \
    .load("path/to/file.csv")
```

# 10. READ MULTIPLE FILES/PATHS

```python
df_multiple = spark.read.csv([
    "path/to/file1.csv",
    "path/to/file2.csv",
    "path/to/directory/*.csv"
])
```

# 11. COMMON OPTIONS ACROSS FORMATS

```python
df_common = spark.read \
    .option("pathGlobFilter", "*.csv") \
    .option("recursiveFileLookup", "true") \
    .option("modifiedBefore", "2024-01-01T00:00:00") \
    .option("modifiedAfter", "2023-01-01T00:00:00") \
```
